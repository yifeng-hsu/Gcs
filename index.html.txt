    const arr = array.slice();

      arr[i] = arr[j];

  function gauss(x, mu, sigma, amp){
    const z = (x - mu) / sigma;
    return amp * Math.exp(-0.5 * z * z);
  }
  function wrap01(x){ x = x % 1; return x < 0 ? x + 1 : x; }
  function baseline(t, amp){ return Math.sin(t * 2 * Math.PI * 0.22) * amp; }
  function qrsTTemplate(ph, shift, scale){
    const sc = (typeof scale === 'number') ? scale : 1;
    return gauss(ph, 0.285 + shift, 0.010, -0.16 * sc) + gauss(ph, 0.305 + shift, 0.012, 1.00 * sc) + gauss(ph, 0.328 + shift, 0.014, -0.26 * sc) + gauss(ph, 0.55 + shift, 0.070, 0.30 * sc);
  }
  function pTemplate(ph, at, scale){
    const sc = (typeof scale === 'number') ? scale : 1;
    return gauss(ph, at, 0.025, 0.17 * sc);
  }

  function nsrSample(t){ const ph = wrap01(t * 72 / 60); return baseline(t, 0.025) + pTemplate(ph, 0.16) + qrsTTemplate(ph, 0, 1); }
  function sssSample(t){
  // 洞不全症候群：3拍がかなり近い間隔で出たあと長いpause
  const cycle = 8.0;
  const ph = wrap01(t / cycle);

  // 最初の25%の中に3拍を詰めて配置
  if (ph > 0.25) {
    return 0; // 長い停止
  }

  const beatWidth = 0.25 / 3;
  const beatIndex = Math.floor(ph / beatWidth);
  const beatStart = beatIndex * beatWidth;
  const local = (ph - beatStart) / beatWidth;

  let v = 0;

  // 小さいP波
  v += gauss(local, 0.10, 0.016, 0.09);

  // QRS
  v += gauss(local, 0.22, 0.010, -0.12);
  v += gauss(local, 0.24, 0.012, 0.90);
  v += gauss(local, 0.265, 0.014, -0.22);

  // T波
  v += gauss(local, 0.42, 0.055, 0.20);

  return v;
}
  function avb1Sample(t){ const ph = wrap01(t * 58 / 60); return baseline(t, 0.020) + pTemplate(ph, 0.12) + qrsTTemplate(ph, 0.10, 1); }
  function wenckSample(t){ const cycle = 4.0; const local = (t % cycle) / cycle; const pPos = [0.10, 0.40, 0.80]; const pr = [0.11, 0.18, null]; let v = 0; for(let i = 0; i < 3; i++){ const p = pPos[i]; v += gauss(local, p, 0.007, 0.04); if(pr[i] !== null){ const qrs = p + pr[i]; v += gauss(local, qrs - 0.010, 0.0055, -0.12) + gauss(local, qrs, 0.0075, 1.02) + gauss(local, qrs + 0.012, 0.0085, -0.28) + gauss(local, qrs + 0.11, 0.025, 0.14); } } return v; }
  function avb2Sample(t){ const cycle = 1.35; const ph = wrap01(t / cycle); const twoCycle = wrap01(t / (cycle * 2)); let v = baseline(t, 0.022) + gauss(ph, 0.14, 0.022, 0.15); if(!(twoCycle > 0.50)){ v += gauss(ph, 0.30, 0.010, -0.16) + gauss(ph, 0.32, 0.012, 0.95) + gauss(ph, 0.345, 0.014, -0.26) + gauss(ph, 0.55, 0.070, 0.26); } return v; }
  function chbSample(t){ const pPh = wrap01(t * 78 / 60); const qPh = wrap01(t * 34 / 60 + 0.12); return baseline(t, 0.018) + gauss(pPh, 0.12, 0.018, 0.12) + gauss(qPh, 0.24, 0.012, -0.14) + gauss(qPh, 0.26, 0.016, 0.82) + gauss(qPh, 0.29, 0.016, -0.22) + gauss(qPh, 0.52, 0.075, 0.22); }
  function pacSample(t){ const ph = wrap01(t * 72 / 60); let v = baseline(t, 0.018) + pTemplate(ph, 0.16, 0.95) + qrsTTemplate(ph, 0, 0.95); if(ph > 0.56 && ph < 0.78){ const local = (ph - 0.56) / 0.22; v += gauss(local, 0.10, 0.016, 0.11) + gauss(local, 0.225, 0.010, -0.14) + gauss(local, 0.245, 0.012, 0.92) + gauss(local, 0.268, 0.014, -0.22) + gauss(local, 0.47, 0.055, 0.18); } return v; }
  function pvcSample(t){ const ph = wrap01(t * 72 / 60); if(ph > 0.62 && ph < 0.90){ const local = (ph - 0.62) / 0.28; return baseline(t, 0.016) + gauss(local, 0.24, 0.020, 1.28) + gauss(local, 0.40, 0.050, -0.82) + gauss(local, 0.70, 0.100, 0.08); } return baseline(t, 0.022) + pTemplate(ph, 0.16, 0.9) + qrsTTemplate(ph, 0, 0.95); }
  function nsvtSample(t){ const runCycle = 5.5; const phRun = wrap01(t / runCycle); if(phRun > 0.32 && phRun < 0.68){ const local = (phRun - 0.32) / 0.36; const vtph = wrap01(local * 4.2); return baseline(t, 0.014) + gauss(vtph, 0.22, 0.032, 1.10) + gauss(vtph, 0.34, 0.038, -0.55) + gauss(vtph, 0.62, 0.080, 0.10); } const ph = wrap01(t * 68 / 60); return baseline(t, 0.020) + gauss(ph, 0.16, 0.024, 0.15) + gauss(ph, 0.285, 0.010, -0.16) + gauss(ph, 0.305, 0.012, 0.92) + gauss(ph, 0.328, 0.014, -0.24) + gauss(ph, 0.55, 0.065, 0.26); }
  function vtSample(t){ const ph = wrap01(t * 180 / 60); return baseline(t, 0.010) + gauss(ph, 0.18, 0.030, 1.18) + gauss(ph, 0.33, 0.040, -0.62) + gauss(ph, 0.60, 0.080, 0.10); }
  function vfSample(t){ return 0.28 * Math.sin(t * 2 * Math.PI * 5.2) + 0.18 * Math.sin(t * 2 * Math.PI * 7.1 + 0.8) + 0.10 * Math.sin(t * 2 * Math.PI * 9.7 + 1.6); }
  function steSample(t){ const ph = wrap01(t * 68 / 60); let v = 0; v += gauss(ph, 0.15, 0.018, 0.09) + gauss(ph, 0.285, 0.010, -0.10) + gauss(ph, 0.305, 0.012, 1.02) + gauss(ph, 0.338, 0.020, -1.02); const start = 0.345; const end = 0.79; if(ph > start && ph < end){ const u = (ph - start) / (end - start); const alpha = 2.1; const beta = 2.8; const hump = Math.pow(u, alpha - 1) * Math.pow(1 - u, beta - 1); v += 1.9 * hump; } return v; }
  function aaiSample(t){ const ph = wrap01(t * 60 / 60); return baseline(t, 0.02) + gauss(ph, 0.145, 0.020, 0.15) + gauss(ph, 0.295, 0.009, -0.15) + gauss(ph, 0.31, 0.012, 0.95) + gauss(ph, 0.332, 0.014, -0.24) + gauss(ph, 0.55, 0.065, 0.28); }
  function vviSample(t){ const ph = wrap01(t * 60 / 60); return baseline(t, 0.018) + gauss(ph, 0.225, 0.014, -0.10) + gauss(ph, 0.245, 0.024, 1.10) + gauss(ph, 0.29, 0.022, -0.45) + gauss(ph, 0.55, 0.085, 0.20); }
  function dddSample(t){ const ph = wrap01(t * 70 / 60); return baseline(t, 0.018) + gauss(ph, 0.125, 0.017, 0.14) + gauss(ph, 0.27, 0.016, -0.10) + gauss(ph, 0.295, 0.024, 1.12) + gauss(ph, 0.34, 0.022, -0.40) + gauss(ph, 0.56, 0.085, 0.22); }

  function sampleScenario(t, id){
    switch(id){
      case 'NSR': return nsrSample(t);
      case 'PAC': return pacSample(t);
      case 'SSS': return sssSample(t);
      case 'AVB1': return avb1Sample(t);
      case 'WENCK': return wenckSample(t);
      case 'AVB2': return avb2Sample(t);
      case 'CHB': return chbSample(t);
      case 'PVC': return pvcSample(t);
      case 'NSVT': return nsvtSample(t);
      case 'VT': return vtSample(t);
      case 'VF': return vfSample(t);
      case 'STE': return steSample(t);
      case 'AAI': return aaiSample(t);
      case 'VVI': return vviSample(t);
      case 'DDD': return dddSample(t);
      default: return nsrSample(t);
    }
  }

  function spikePositionsForScenario(id, t){
    if(id === 'AAI'){
      const ph = wrap01(t);
      return Math.abs(ph - 0.108) < 0.002 ? [1] : [];
    }
    if(id === 'VVI'){
      const ph = wrap01(t);
      return Math.abs(ph - 0.19) < 0.002 ? [1] : [];
    }
    if(id === 'DDD'){
      const ph = wrap01(t * 70 / 60);
      const out = [];
      if(Math.abs(ph - 0.10) < 0.002) out.push(1);
      if(Math.abs(ph - 0.24) < 0.002) out.push(1);
      return out;
    }
    return [];
  }

  function pushSamples(dt){
    const count = Math.max(1, Math.floor(dt * fs));
    for(let i = 0; i < count; i++){
      engineT += 1 / fs;
      buffer[writeIdx] = sampleScenario(engineT, current.id);
      writeIdx = (writeIdx + 1) % N;
    }
  }

  function drawGrid(){
    ctx.clearRect(0, 0, W, H);
    ctx.fillStyle = '#050807';
    ctx.fillRect(0, 0, W, H);
    const mm = 4;
    ctx.lineWidth = 1;
    ctx.strokeStyle = 'rgba(53,255,154,0.05)';
    for(let x = 0; x <= W; x += mm){ ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, H); ctx.stroke(); }
    for(let y = 0; y <= H; y += mm){ ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(W, y); ctx.stroke(); }
    ctx.strokeStyle = 'rgba(53,255,154,0.10)';
    for(let x = 0; x <= W; x += mm * 5){ ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, H); ctx.stroke(); }
    for(let y = 0; y <= H; y += mm * 5){ ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(W, y); ctx.stroke(); }
  }

  function drawWave(){
    ctx.save();
    ctx.lineWidth = 2;
    ctx.strokeStyle = 'rgba(53,255,154,0.95)';
    ctx.shadowBlur = 8;
    ctx.shadowColor = 'rgba(53,255,154,0.55)';
    ctx.beginPath();
    for(let i = 0; i < N; i++){
      const idx = (writeIdx + i) % N;
      const x = i / (N - 1) * W;
      const y = midY - buffer[idx] * 60;
      if(i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
    }
    ctx.stroke();
    ctx.restore();

    if(current.id === 'AAI' || current.id === 'VVI' || current.id === 'DDD'){
      ctx.strokeStyle = '#ffffff';
      ctx.lineWidth = 1.6;
      for(let i = 0; i < N; i++){
        const x = i / (N - 1) * W;
        const t = engineT - ((N - i) / fs);
        const spikes = spikePositionsForScenario(current.id, t);
        if(spikes.length){
          const top = midY - 112;
          const bottom = midY - 96;
          ctx.beginPath();
          ctx.moveTo(x, top);
          ctx.lineTo(x, bottom);
          ctx.stroke();
        }
      }
    }

    const grad = ctx.createLinearGradient(W - 90, 0, W, 0);
    grad.addColorStop(0, 'rgba(53,255,154,0)');
    grad.addColorStop(1, 'rgba(53,255,154,0.20)');
    ctx.fillStyle = grad;
    ctx.fillRect(W - 90, 0, 90, H);
  }

  function animate(){
    const now = performance.now() / 1000;
    const dt = Math.min(0.05, now - lastNow);
    lastNow = now;
    pushSamples(dt);
    drawGrid();
    drawWave();
    requestAnimationFrame(animate);
  }

  function runSelfTests(){
    console.assert(Array.isArray(activeQuizIds) && activeQuizIds.length > 0, 'activeQuizIds should be populated');
    console.assert(typeof steSample(0.1) === 'number', 'steSample should return number');
    console.assert(typeof vtSample(0.1) === 'number', 'vtSample should return number');
    console.assert(typeof vfSample(0.1) === 'number', 'vfSample should return number');
    console.assert(randomChoices('NSR').length === 3, 'randomChoices should return 3 choices');
  }

  populateUI();
  setTrainingMode('basic');
  setScenario('NSR');
  animate();
  runSelfTests();
})();
